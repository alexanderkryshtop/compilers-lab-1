            concat: str = "+",
            kleene_star: str = "*",
            opening_bracket: str = "(",

    ):
        self.letters = [chr(c) for c in range(ord("a"), ord("z") + 1)]
        self.epsilon = epsilon
        self.union = union
        self.concat = concat
        self.kleene_star = kleene_star
        self.opening_bracket = opening_bracket
        self.closing_bracket = closing_bracket
